public static String appendFromArray(String string, String[] array, int arrayIndex, int amountOfTimes){
	if (amountOfTimes < 1 || string == null) {
	  return string;
	}
	for (int i = 0; i < amountOfTimes; i++) {
	  string = string + array[arrayIndex];
	}
	return string;
}
